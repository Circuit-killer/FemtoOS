142 #if (cfgSysSqueezeState == cfgFalse)
143  /* The devAuxSysRegBit is assumed to be zero upon entry, signaling per default that the
144  * interrupts are switched off. The value of the I bit is copied to the ARB bit. */
145  #define portResqueGlobalInterruptState() \
146  asm volatile ( "brid 67f \n\t" \
147  "cli \n\t" \
148  "sbi %[_ASR_],%[_ARB_] \n\t" \
149  "67: " :: \
150  [_ARB_] "i" (devAuxSysRegBit), \
151  [_ASR_] "i" (_SFR_IO_ADDR(devAuxSysReg)))
152 #else
153  /* We signal with the H bit being zero that we originate from a manual Yield. The old
154  * value of the I bit is copied to the H bit and the I bit is cleaned. The code below is
155  * tricky. First we clean the H bit, and subsequently test if interrupts are enabled. If
156  * so we must set the H bit and clean the I bit. The rest is unimportant, since we came from
157  * a manual call. To save we first clean H and test for I afterwards. But what if an interrupt
158  * comes after cleaning H and before the test? H could be set again when we return, since
159  * the H bit may be not conserved due to our own approach. The point is, it does not matter. If
160  * I was cleaned before, interrupts are not possible so H stays reset, and if I was not,
161  * H will be set anyway. So we do not fear an interrupt here.
162  * */
163  #define portResqueGlobalInterruptState() \
164  asm volatile ( "clh \n\t" /* Clear H bit */ \
165  "brid 67f \n\t" \
166  "cli \n\t" /* Clean I bit */ \
167  "seh \n\t" /* Set H bit */ \
168  "67: " :: \
169  [_Hb_] "i" (preBitSet1(0x00,SREG_H)) )
170 #endif
